        DclFld SelectedItem Type(ComboboxItem)

        SelectedItem = comboboxFormType.SelectedItem *As ComboboxItem 
        If SelectedItem.Value = MODAL
            ShowModalForm()
        Else 
            ShowNonModalForm()
        EndIf  
    EndSr

        DclFld NonModalForm Type(formNonModal) New()

        dr = NonModalForm.ShowDialog()
